\begin{denotation}[4.6cm]
  \item[ANN] Approximate Nearest Neighbor，近似最近邻
  \item[AUC] Area Under Curve，曲线下面积
  \item[BTO] Basic Tensor Operator，基础张量算子
  \item[FAISS] Facebook AI Similarity Search
  \item[FO] Functional Operator，功能算子
  \item[GPU] Graphics Processing Unit，图形处理器
  \item[KNN] K-Nearest Neighbors，$k$ 近邻
  \item[PR-AUC] Precision-Recall Area Under Curve
  \item[QPS] Queries Per Second，每秒查询数
  \item[RAPIDS] 面向 GPU 数据科学与分析计算的开源加速生态
  \item[ROC-AUC] Receiver Operating Characteristic Area Under Curve
  \item[TOD] Tensor Operations for Outlier Detection
  \item[Top-$k$ overlap] 近似排序结果与基线排序结果的前 $k$ 重合度
  \item[\texttt{adaptive\_ratio}] 多 GPU 动态分流比例策略
  \item[bounded batching] 在时延约束下受控执行的微批处理策略
  \item[rerank] 对粗排候选执行更高精度精排计算的步骤
\end{denotation}
